  <script>
    // Dinamik blog yazıları
    const blogData = [
      { "title": "İlk Yazım", "date": "2025-09-20", "content": "Buraya blog yazınızı ekleyebilirsiniz. Eğitim, deneyim veya ilgi alanlarınızı paylaşabilirsiniz." },
      { "title": "İkinci Yazım", "date": "2025-09-21", "content": "İkinci yazınızı buraya ekleyebilirsiniz. Artık JSON ile yönetmek çok kolay." }
    ];

    const blogContainer = document.getElementById('blogPosts');
    blogData.forEach(post => {
      const postDiv = document.createElement('div');
      postDiv.classList.add('post');
      postDiv.innerHTML = `<h3>${post.title}</h3><div class="date">${post.date}</div><p>${post.content}</p>`;
      blogContainer.appendChild(postDiv);
    });

    // Blog animasyonu
    const posts = document.querySelectorAll('.post');
    const observer = new IntersectionObserver(entries => {
      entries.forEach(entry => {
        if(entry.isIntersecting) entry.target.classList.add('show');
      });
    }, { threshold: 0.1 });
    posts.forEach(post => observer.observe(post));

    // Form işleme
    const form = document.getElementById('contactForm');
    const thankYou = document.getElementById('thankYouMessage');
    form.addEventListener('submit', function(e){
      e.preventDefault();
      const formData = new FormData(form);
      fetch(form.action, {
        method: form.method,
        body: formData,
        headers: { 'Accept':'application/json' }
      })
      .then(response => {
        if(response.ok){
          form.reset();
          thankYou.style.display = 'block';
          setTimeout(()=>{ thankYou.style.display='none'; },5000);
        } else { alert('Form gönderilirken hata oluştu.'); }
      }).catch(()=>alert('Form gönderilirken hata oluştu.'));
    });
  </script>
